def identify_pareto(scores):

    pareto_index = []

    for i, (c1, c2) in enumerate(scores):

        dominated = any((c1 < other_c1 and c2 < other_c2) for j, (other_c1, other_c2) in enumerate(scores) if i != j)
        if not dominated:
            pareto_index.append(i)
    return pareto_index


accuracy = cv_results['mean_test_accuracy_youden']
precision = cv_results['mean_test_precision_youden']


scores = list(zip(accuracy, precision))


pareto_indices = identify_pareto(scores)


pareto_data = {
    'Mean Test Accuracy': [accuracy[i] for i in pareto_indices],
    'Mean Test Precision': [precision[i] for i in pareto_indices],
    'Parameters': [str(cv_results['params'][i]) for i in pareto_indices]
}


pareto_fig = px.scatter(
    pareto_data,
    x='Mean Test Accuracy',
    y='Mean Test Precision',
    hover_data=['Parameters'],
    title='Pareto Front of Precision vs. Accuracy',
    labels={'Mean Test Accuracy': 'Accuracy', 'Mean Test Precision': 'Precision'}
)


pareto_fig.show()
